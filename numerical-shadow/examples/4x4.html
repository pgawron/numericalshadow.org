\[ A^{(4)}_4= \begin{pmatrix} 1 & 0 & 0 & 1 \\ 0 & \mathrm{i} & 0 & 0 \\ 0 & 0 & -1 & 0 \\ 0 & 0 & 0 & -\mathrm{i} \end{pmatrix}, \] \[ A^{(4)}_5= \begin{pmatrix} \mathrm{i} & 0 & -1 & 0 \\ 0 & 0 & -1 & 0 \\ 1 & 1 & 1-\mathrm{i} & 0 \\ 0 & 0 & 0 & 1 \end{pmatrix}, \]
\[ A^{(4)}_6= \begin{pmatrix} 1 & 0 & 1 & 0 \\ 0 & \mathrm{i} & 0 & 1 \\ 0 & 0 & -1 & 0 \\ 0 & 0 & 0 & -\mathrm{i} \end{pmatrix}, \] \[ A^{(4)}_7= \begin{pmatrix} 1 & 0 & 0 & 0 \\ 0 & \mathrm{i} & 0 & 1 \\ 0 & 0 & -1 & 0 \\ 0 & 0 & 0 & -\mathrm{i} \end{pmatrix}, \]
